    <!-- Sound Effects (Using Audio API to avoid external assets) -->

    <script>

        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();

        function playPopSound() {

            const oscillator = audioCtx.createOscillator();

            const gainNode = audioCtx.createGain();

            oscillator.type = 'sine';

            oscillator.frequency.setValueAtTime(500, audioCtx.currentTime);

            oscillator.frequency.exponentialRampToValueAtTime(100, audioCtx.currentTime + 0.1);

            gainNode.gain.setValueAtTime(0.2, audioCtx.currentTime);

            gainNode.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.1);

            oscillator.connect(gainNode);

            gainNode.connect(audioCtx.destination);

            oscillator.start();

            oscillator.stop(audioCtx.currentTime + 0.1);

        }


        // Game State

        let score = 0;

        let timeLeft = 30;

        let gameActive = false;

        let timerInterval;

        let spawnInterval;

        const GRID_SIZE = 20;


        const introScreen = document.getElementById('intro-screen');

        const gameScreen = document.getElementById('game-screen');

        const resultScreen = document.getElementById('result-screen');

        const gameGrid = document.getElementById('game-grid');

        const timerDisplay = document.getElementById('timer');

        const scoreDisplay = document.getElementById('score');


        // SVG Icons

        const bulbOnSvg = `<svg viewBox="0 0 24 24" fill="#facc15" class="w-full h-full bulb-on"><path d="M12 2C8.13 2 5 5.13 5 9c0 2.38 1.19 4.47 3 5.74V17c0 .55.45 1 1 1h6c.55 0 1-.45 1-1v-2.26c1.81-1.27 3-3.36 3-5.74 0-3.87-3.13-7-7-7zM9 21c0 .55.45 1 1 1h4c.55 0 1-.45 1-1v-1H9v1z"/></svg>`;

        const bulbOffSvg = `<svg viewBox="0 0 24 24" fill="#9ca3af" class="w-full h-full"><path d="M12 2C8.13 2 5 5.13 5 9c0 2.38 1.19 4.47 3 5.74V17c0 .55.45 1 1 1h6c.55 0 1-.45 1-1v-2.26c1.81-1.27 3-3.36 3-5.74 0-3.87-3.13-7-7-7zM9 21c0 .55.45 1 1 1h4c.55 0 1-.45 1-1v-1H9v1z"/></svg>`;


        function initGrid() {

            gameGrid.innerHTML = '';

            for (let i = 0; i < GRID_SIZE; i++) {

                const bulb = document.createElement('div');

                bulb.className = 'light-bulb bg-gray-100 p-2 rounded-xl h-20 flex items-center justify-center';

                bulb.dataset.id = i;

                bulb.dataset.status = 'off';

                bulb.innerHTML = bulbOffSvg;

                gameGrid.appendChild(bulb);

            }

        }


        function startGame() {

            score = 0;

            timeLeft = 30;

            gameActive = true;

            

            introScreen.classList.remove('active');

            resultScreen.classList.remove('active');

            gameScreen.classList.add('active');

            

            scoreDisplay.innerText = score;

            timerDisplay.innerText = timeLeft;

            

            initGrid();

            

            timerInterval = setInterval(() => {

                timeLeft--;

                timerDisplay.innerText = timeLeft;

                if (timeLeft <= 0) endGame();

            }, 1000);


            // Logic to turn on bulbs randomly

            spawnInterval = setInterval(() => {

                if (!gameActive) return;

                const bulbs = document.querySelectorAll('.light-bulb');

                const randomIndex = Math.floor(Math.random() * bulbs.length);

                const target = bulbs[randomIndex];

                

                if (target.dataset.status === 'off') {

                    target.dataset.status = 'on';

                    target.innerHTML = bulbOnSvg;

                    target.classList.add('bg-yellow-50');

                    target.onclick = () => turnOff(target);

                    

                    // Automatically turn off if not clicked after 1.5s

                    setTimeout(() => {

                        if (target.dataset.status === 'on' && gameActive) {

                            resetBulb(target);

                        }

                    }, 1500);

                }

            }, 600);

        }


        function turnOff(element) {

            if (element.dataset.status !== 'on' || !gameActive) return;

            

            playPopSound();

            score++;

            scoreDisplay.innerText = score;

            

            element.dataset.status = 'done'; // Prevent double click

            element.innerHTML = bulbOffSvg;

            element.classList.remove('bg-yellow-50');

            element.classList.add('bg-green-100');

            

            // Animation for feedback

            element.style.transform = 'scale(0.8)';

            setTimeout(() => {

                element.style.transform = 'scale(1)';

                resetBulb(element);

            }, 200);

        }


        function resetBulb(element) {

            element.dataset.status = 'off';

            element.innerHTML = bulbOffSvg;

            element.classList.remove('bg-yellow-50', 'bg-green-100');

            element.onclick = null;

        }


        function endGame() {

            gameActive = false;

            clearInterval(timerInterval);

            clearInterval(spawnInterval);

            

            // Calculate results

            const co2 = score * 0.5;

            const trees = (co2 / 2.5).toFixed(1);

            

            document.getElementById('final-score').innerText = score;

            document.getElementById('res-hours').innerText = score;

            document.getElementById('res-co2').innerText = co2;

            document.getElementById('res-trees').innerText = trees;

            

            gameScreen.classList.remove('active');

            resultScreen.classList.add('active');

        }


        function restartGame() {

            resultScreen.classList.remove('active');

            introScreen.classList.add('active');

        }
