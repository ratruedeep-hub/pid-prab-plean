        // Audio Logic

        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();

        function playPopSound() {

            const osc = audioCtx.createOscillator();

            const gain = audioCtx.createGain();

            osc.type = 'sine';

            osc.frequency.setValueAtTime(600, audioCtx.currentTime);

            osc.frequency.exponentialRampToValueAtTime(100, audioCtx.currentTime + 0.1);

            gain.gain.setValueAtTime(0.3, audioCtx.currentTime);

            gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.1);

            osc.connect(gain);

            gain.connect(audioCtx.destination);

            osc.start();

            osc.stop(audioCtx.currentTime + 0.1);

        }


        let score = 0;

        let timeLeft = 30;

        let gameActive = false;

        let timerInterval;

        let spawnInterval;

        const GRID_SIZE = 20;


        const bulbOnSvg = `<svg viewBox="0 0 24 24" fill="#facc15" class="w-12 h-12 bulb-on"><path d="M12 2C8.13 2 5 5.13 5 9c0 2.38 1.19 4.47 3 5.74V17c0 .55.45 1 1 1h6c.55 0 1-.45 1-1v-2.26c1.81-1.27 3-3.36 3-5.74 0-3.87-3.13-7-7-7zM9 21c0 .55.45 1 1 1h4c.55 0 1-.45 1-1v-1H9v1z"/></svg>`;

        const bulbOffSvg = `<svg viewBox="0 0 24 24" fill="#d1d5db" class="w-10 h-10"><path d="M12 2C8.13 2 5 5.13 5 9c0 2.38 1.19 4.47 3 5.74V17c0 .55.45 1 1 1h6c.55 0 1-.45 1-1v-2.26c1.81-1.27 3-3.36 3-5.74 0-3.87-3.13-7-7-7zM9 21c0 .55.45 1 1 1h4c.55 0 1-.45 1-1v-1H9v1z"/></svg>`;


        function initGrid() {

            const grid = document.getElementById('game-grid');

            grid.innerHTML = '';

            for (let i = 0; i < GRID_SIZE; i++) {

                const bulb = document.createElement('div');

                bulb.className = 'light-bulb bg-white rounded-2xl h-24 flex items-center justify-center shadow-sm border border-gray-100';

                bulb.dataset.status = 'off';

                bulb.innerHTML = bulbOffSvg;

                grid.appendChild(bulb);

            }

        }


        function startGame() {

            score = 0;

            timeLeft = 30;

            gameActive = true;

            

            document.getElementById('intro-screen').classList.remove('active');

            document.getElementById('result-screen').classList.remove('active');

            document.getElementById('game-screen').classList.add('active');

            

            document.getElementById('score').innerText = score;

            document.getElementById('timer').innerText = timeLeft;

            

            initGrid();

            

            timerInterval = setInterval(() => {

                timeLeft--;

                document.getElementById('timer').innerText = timeLeft;

                if (timeLeft <= 0) endGame();

            }, 1000);


            spawnInterval = setInterval(() => {

                if (!gameActive) return;

                const bulbs = document.querySelectorAll('.light-bulb');

                const available = Array.from(bulbs).filter(b => b.dataset.status === 'off');

                if (available.length === 0) return;


                const target = available[Math.floor(Math.random() * available.length)];

                activateBulb(target);

            }, 550);

        }


        function activateBulb(element) {

            element.dataset.status = 'on';

            element.innerHTML = bulbOnSvg;

            element.classList.add('bg-yellow-50', 'border-yellow-200');

            element.onclick = () => {

                if (element.dataset.status === 'on') {

                    playPopSound();

                    score++;

                    document.getElementById('score').innerText = score;

                    element.dataset.status = 'off';

                    element.innerHTML = bulbOffSvg;

                    element.classList.remove('bg-yellow-50', 'border-yellow-200');

                    element.classList.add('bg-green-50');

                    setTimeout(() => element.classList.remove('bg-green-50'), 300);

                }

            };


            // Auto-off

            setTimeout(() => {

                if (element.dataset.status === 'on' && gameActive) {

                    element.dataset.status = 'off';

                    element.innerHTML = bulbOffSvg;

                    element.classList.remove('bg-yellow-50', 'border-yellow-200');

                }

            }, 1200);

        }


        function endGame() {

            gameActive = false;

            clearInterval(timerInterval);

            clearInterval(spawnInterval);

            

            const co2 = score * 0.5;

            const trees = (co2 / 2.5).toFixed(1);

            

            document.getElementById('final-score').innerText = score;

            document.getElementById('res-hours').innerText = score;

            document.getElementById('res-co2').innerText = co2;

            document.getElementById('res-trees').innerText = trees;

            

            document.getElementById('game-screen').classList.remove('active');

            document.getElementById('result-screen').classList.add('active');

        }


        function restartGame() {

            document.getElementById('result-screen').classList.remove('active');

            document.getElementById('intro-screen').classList.add('active');

        }
